## 用法

```tsx
import {
  Collapsible,
  CollapsibleContent,
  CollapsibleTrigger,
} from "@/components/ui/collapsible"
```

```tsx
<Collapsible>
  <CollapsibleTrigger>我可以在我的项目中使用这个吗？</CollapsibleTrigger>
  <CollapsibleContent>
    是的。可以自由用于个人和商业项目。不需要署名。
  </CollapsibleContent>
</Collapsible>
```
