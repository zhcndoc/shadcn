## 使用

```tsx
import {
  Select,
  SelectContent,
  SelectItem,
  SelectTrigger,
  SelectValue,
} from "@/components/ui/select"
```

```tsx
<Select>
  <SelectTrigger className="w-[180px]">
    <SelectValue placeholder="主题" />
  </SelectTrigger>
  <SelectContent>
    <SelectItem value="light">亮色</SelectItem>
    <SelectItem value="dark">暗色</SelectItem>
    <SelectItem value="system">系统</SelectItem>
  </SelectContent>
</Select>
```
