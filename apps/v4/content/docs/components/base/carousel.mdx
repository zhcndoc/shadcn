### 间距

要设置项目之间的间距，我们在 `<CarouselItem />` 上使用 `pl-[VALUE]` 工具类，在 `<CarouselContent />` 上使用负的 `-ml-[VALUE]`。

<Callout className="mt-6">

您也可以根据自己的项目需求调整该设置。

</Callout>

## API

使用状态和 `setApi` 属性获取轮播图 API 实例。

<ComponentPreview
  styleName="base-nova"
  name="carousel-api"
  title="轮播图"
  description="一个带幻灯片计数器的轮播图。"
/>

```tsx showLineNumbers {1,4,22}
import { type CarouselApi } from "@/components/ui/carousel"

export function Example() {
  const [api, setApi] = React.useState<CarouselApi>()
  const [current, setCurrent] = React.useState(0)
  const [count, setCount] = React.useState(0)

  React.useEffect(() => {
    if (!api) {
      return
    }

    setCount(api.scrollSnapList().length)
    setCurrent(api.selectedScrollSnap() + 1)

    api.on("select", () => {
      setCurrent(api.selectedScrollSnap() + 1)
    })
  }, [api])

  return (
    <Carousel setApi={setApi}>
      <CarouselContent>
        <CarouselItem>...</CarouselItem>
        <CarouselItem>...</CarouselItem>
        <CarouselItem>...</CarouselItem>
      </CarouselContent>
    </Carousel>
  )
}
```

## 事件

您可以使用从 `setApi` 获得的 api 实例监听事件。

```tsx showLineNumbers {1,4-14,16}
import { type CarouselApi } from "@/components/ui/carousel"

export function Example() {
  const [api, setApi] = React.useState<CarouselApi>()

  React.useEffect(() => {
    if (!api) {
      return
    }

    api.on("select", () => {
      // 在选中时执行某些操作。
    })
  }, [api])

  return (
    <Carousel setApi={setApi}>
      <CarouselContent>
        <CarouselItem>...</CarouselItem>
        <CarouselItem>...</CarouselItem>
        <CarouselItem>...</CarouselItem>
      </CarouselContent>
    </Carousel>
  )
}
```

更多关于事件的使用信息，请参见 [Embla Carousel 文档](https://www.embla-carousel.com/api/events/)。

## 插件

您可以通过 `plugins` 属性给轮播图添加插件。

```ts showLineNumbers {1,6-10}
import Autoplay from "embla-carousel-autoplay"

export function Example() {
  return (
    <Carousel
      plugins={[
        Autoplay({
          delay: 2000,
        }),
      ]}
    >
      // ...
    </Carousel>
  )
}
```

## RTL

要在 shadcn/ui 中启用 RTL（从右到左）支持，请参见 [RTL 配置指南](/docs/rtl)。

<ComponentPreview
  styleName="base-nova"
  name="carousel-rtl"
  direction="rtl"
  previewClassName="h-80 sm:h-[32rem]"
/>

在为 RTL 语言本地化轮播图时，您需要在 `opts` 属性中设置 `direction` 选项以匹配文本方向。这样可以确保轮播图按正确方向滚动。

```tsx showLineNumbers {2-5}
<Carousel
  dir={dir}
  opts={{
    direction: dir,
  }}
>
  <CarouselContent>
    <CarouselItem>...</CarouselItem>
    <CarouselItem>...</CarouselItem>
    <CarouselItem>...</CarouselItem>
  </CarouselContent>
  <CarouselPrevious className="rtl:rotate-180" />
  <CarouselNext className="rtl:rotate-180" />
</Carousel>
```

`direction` 选项接受 `"ltr"` 或 `"rtl"`，应与 `dir` 属性值匹配。您可能还需要使用 `rtl:rotate-180` 类旋转导航按钮，以确保它们指向正确方向。
