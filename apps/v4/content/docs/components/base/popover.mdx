## 用法

```tsx showLineNumbers
import {
  Popover,
  PopoverContent,
  PopoverDescription,
  PopoverHeader,
  PopoverTitle,
  PopoverTrigger,
} from "@/components/ui/popover"
```

```tsx showLineNumbers
<Popover>
  <PopoverTrigger render={<Button variant="outline" />}>
    打开弹出框
  </PopoverTrigger>
  <PopoverContent>
    <PopoverHeader>
      <PopoverTitle>标题</PopoverTitle>
      <PopoverDescription>这里是描述文本。</PopoverDescription>
    </PopoverHeader>
  </PopoverContent>
</Popover>
```
