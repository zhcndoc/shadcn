## 使用方法

```tsx showLineNumbers
import {
  Tooltip,
  TooltipContent,
  TooltipTrigger,
} from "@/components/ui/tooltip"
```

```tsx showLineNumbers
<Tooltip>
  <TooltipTrigger>悬停</TooltipTrigger>
  <TooltipContent>
    <p>添加到库</p>
  </TooltipContent>
</Tooltip>
```
