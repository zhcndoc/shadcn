## 使用示例

```tsx showLineNumbers
import {
  Select,
  SelectContent,
  SelectItem,
  SelectTrigger,
  SelectValue,
} from "@/components/ui/select"
```

```tsx showLineNumbers
<Select>
  <SelectTrigger className="w-[180px]">
    <SelectValue placeholder="主题" />
  </SelectTrigger>
  <SelectContent>
    <SelectItem value="light">明亮</SelectItem>
    <SelectItem value="dark">暗黑</SelectItem>
    <SelectItem value="system">系统</SelectItem>
  </SelectContent>
</Select>
```
